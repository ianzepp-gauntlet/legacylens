\section{Phase 1: Constraints}

\subsection{Scale \& Load Profile}

\begin{description}[style=nextline, leftmargin=2em]
  \item[Target codebase] AWS CardDemo (\texttt{aws-mainframe-modernization-carddemo})
    \begin{itemize}[nosep]
      \item Credit card management mainframe application (COBOL/CICS/VSAM/JCL)
      \item 44 COBOL programs ($\sim$28K LOC), 62 copybooks ($\sim$10.7K LOC), 106 COBOL files total ($\sim$40.8K LOC)
      \item Plus JCL, BMS maps, DDL, assembly --- rich supporting artifacts
      \item Real business logic: accounts, cards, transactions, bill payments, statements
      \item Intentionally varied coding styles for analysis tooling --- ideal for RAG
      \item Apache 2.0 licensed, well-documented
    \end{itemize}
  \item[Exceeds minimum] 10,000+ LOC across 50+ files
  \item[Query volume] Low steady-state (developer + reviewers), spiky from automated parallel evals
  \item[Ingestion] Batch (one-time ingest of static codebase)
  \item[Latency target] $<$3 seconds end-to-end
\end{description}

\subsection{Budget \& Cost Ceiling}

\begin{description}[style=nextline, leftmargin=2em]
  \item[Overall budget] $\sim$\$100/month comfortable
  \item[Embedding costs] No ceiling (negligible for codebase size)
  \item[LLM costs] No ceiling, OpenAI
  \item[Preference] Managed services to save time
\end{description}

\subsection{Time to Ship}

\begin{description}[style=nextline, leftmargin=2em]

\section{Phase 2: Architecture Decisions}

\subsection{Vector Database: Pinecone}

\begin{itemize}[nosep]
  \item Managed cloud, free tier for MVP, paid tier for final submission
  \item Hybrid search (vector + keyword) enabled
  \item Metadata fields: file path, line numbers, section type, function/paragraph name, has\_comments
  \item Comments are high-priority content --- preserve alongside associated code
\end{itemize}

\subsection{Embedding Strategy}

\begin{description}[style=nextline, leftmargin=2em]
  \item[Model] OpenAI text-embedding-3-small (1536 dimensions)
  \item[Fallback/alternative] Can switch to text-embedding-3-large if quality is poor (re-index takes minutes)
  \item[Local toggle] Ollama support for offline dev/testing (configurable provider)
  \item[Batch processing] Optimize later
\end{description}

\subsection{Chunking Approach}

\begin{description}[style=nextline, leftmargin=2em]
  \item[Strategy] Syntax-aware from the start (COBOL DIVISIONs, SECTIONs, PARAGRAPHs)
  \item[Max chunk size] $\sim$1000 tokens, split if exceeded
  \item[Overlap strategy] Deferred until codebase is reviewed
  \item[Fallback] Fixed-size + overlap for anything that doesn't parse cleanly
\end{description}

\subsection{Retrieval Pipeline}

\begin{description}[style=nextline, leftmargin=2em]
  \item[Top-k] 10 (configurable)
  \item[Re-ranking] None for MVP
  \item[Context window management] None for MVP
  \item[Multi-query / query expansion] Deferred (not MVP)
  \item[Dynamic k (LLM-suggested)] Stretch goal
\end{description}

\subsection{Answer Generation}

\begin{description}[style=nextline, leftmargin=2em]
  \item[LLM] OpenAI (GPT-4o and GPT-4o-mini, configurable, test both)
  \item[Prompt design] Keep simple for MVP
  \item[Citations] Inline file/line references, e.g.\ \texttt{(file.cob:42-58)}
  \item[Streaming] No, full response for MVP
\end{description}

\subsection{Framework: LangChain}

\begin{itemize}[nosep]
  \item Pairs with OpenAI + Pinecone
  \item Prior exposure from RAG cookbook prework
\end{itemize}
